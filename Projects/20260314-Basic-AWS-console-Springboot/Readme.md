### Infrastructure
1.1. Create manually 
1.1.1 key-pair
```
KEY_NAME=kp-2026-03-14
aws ec2 create-key-pair \
    --key-name $KEY_NAME \
    --query 'KeyMaterial' \
    --output text > $KEY_NAME.pem
chmod 400 $KEY_NAME.pem
```

1.1.2 security group
```
aws ec2 describe-vpcs \
    --query 'Vpcs[*].{ID:VpcId, Name:Tags[?Key==`Name`]|[0].Value}' \
    --output table 
VPC=vpc-xxx

aws ec2 describe-subnets \
    --filters "Name=vpc-id,Values=$VPC" \
    --query 'Subnets[*].{ID:SubnetId, AZ:AvailabilityZone}' \
    --output table 
SUBNET=subnet-xxx

aws ec2 create-security-group \
    --group-name "2026-03-14-sg" \
    --description "idk" \
    --vpc-id $VPC
SG_ID=sg-xxx

aws ec2 authorize-security-group-ingress \
    --group-id "$SG_ID" \
    --protocol tcp \
    --port 22 \
    --cidr 0.0.0.0/0

aws ec2 authorize-security-group-ingress \
    --group-id "$SG_ID" \
    --protocol tcp \
    --port 80 \
    --cidr 0.0.0.0/0
```

1.1.3 EC2
```
aws ssm get-parameters-by-path \
    --path "/aws/service/ami-amazon-linux-latest" \
    --region eu-central-1 \
    --query "Parameters[?contains(Name, 'al2023-ami-kernel-default-x86_64')].Value" \
    --output text
AMI=ami-096a4fdbcf530d8e0

aws ec2 run-instances \
    --image-id $AMI \
    --count 1 \
    --instance-type t2.micro \
    --key-name $KEY_NAME \
    --security-group-ids $SG_ID \
    --subnet-id $SUBNET \
    --associate-public-ip-address \
    --region eu-central-1 

aws ec2 describe-instances \
    --filters "Name=instance-state-name,Values=running" \
    --query 'Reservations[*].Instances[*].{ID:InstanceId, PublicIP:PublicIpAddress, Name:Tags[?Key==`Name`]|[0].Value}' \
    --output table   
PUBLIC_IP=X.X.X.X    
```

### Infra setup 
```
ssh -i $KEY_NAME.pem  ec2-user@$PUBLIC_IP

sudo dnf update -y
sudo dnf install iptables-services -y
sudo dnf install java-21-amazon-corretto java-21-amazon-corretto-devel -y
java -version
```

```
sudo mkdir -p /opt/demo
sudo chmod -R 777 /opt/demo

ARTEFACT_PATH="app-code/target/demo-0.0.1-SNAPSHOT.jar"
scp -i $KEY_NAME.pem $ARTEFACT_PATH ec2-user@$PUBLIC_IP:/opt/demo/demo.jar

ssh -i $KEY_NAME.pem  ec2-user@$PUBLIC_IP

sudo vi /etc/systemd/system/demo.service
sudo systemctl daemon-reload
sudo systemctl enable demo.service
sudo systemctl restart demo.service
sudo systemctl status demo.service

<!-- port forward 80 to 8080 -->
sudo iptables -t nat -A PREROUTING -p tcp --dport 80 -j REDIRECT --to-port 8080
sudo iptables -t nat -A OUTPUT -o lo -p tcp --dport 80 -j REDIRECT --to-port 8080
```


### Infra delete
```
aws ec2 describe-instances \
    --filters "Name=instance-state-name,Values=running" \
    --query 'Reservations[*].Instances[*].{ID:InstanceId, PublicIP:PublicIpAddress, Name:Tags[?Key==`Name`]|[0].Value}' \
    --output table  
INSTANCE_IDS=i-xxx    
aws ec2 terminate-instances --instance-ids $INSTANCE_IDS
```

```
aws ec2 describe-security-groups \
    --query 'SecurityGroups[*].{Name:GroupName, ID:GroupId, VPC:VpcId}' \
    --output table
$SG_ID=sg-xxx
aws ec2 delete-security-group --group-id $SG_ID 
```

```

    --query 'KeyPairs[*].{Name:KeyName, ID:KeyPairId}' \
    --output table
```
